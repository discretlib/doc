Define Access Rights 

mutate {
    sys.Room {
        admin: [{
            verif_key: $peer_key
        }]

        authorisations: [{
            name: "house"
            rights:[{
                entity: "Person"
                mutate_self: true
                mutate_all: true
            },{
                entity: "Pet"
                mutate_self: true
                mutate_all: false
            }]
        }]
    }
}
